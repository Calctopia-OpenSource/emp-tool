# install steps
macro (install_dir _dir _patterns)
	FOREACH(pattern ${_patterns})
		install(DIRECTORY ${CMAKE_SOURCE_DIR}/${_dir} DESTINATION include FILES_MATCHING PATTERN ${pattern})
	ENDFOREACH(pattern)
endmacro()
